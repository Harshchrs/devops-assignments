# 0. Go to home directory
cd ~

# 1. Create parent folder for all assignments
mkdir devops-assignments

# 2. Step 1: Creating and Renaming Files/Directories
mkdir test_dir
cd test_dir
touch example.txt
mv example.txt renamed_example.txt
ls    # verify

# 3. Step 2: Viewing File Contents
cat /etc/passwd
head -n 5 /etc/passwd
tail -n 5 /etc/passwd

# 4. Step 3: Searching for Patterns
grep "root" /etc/passwd

# 5. Step 4: Zipping and Unzipping
cd ..
zip -r test_dir.zip test_dir
mkdir unzipped_dir
unzip test_dir.zip -d unzipped_dir

# Optional: single file zip example
cd unzipped_dir
zip single_file.zip test_dir/renamed_example.txt
mkdir new_test_unzip
unzip single_file.zip -d new_test_unzip

# 6. Step 5: Downloading Files
# Make sure you are in unzipped_dir
pwd   # check current directory
wget https://raw.githubusercontent.com/github/gitignore/main/Python.gitignore
ls    # verify downloaded file

# 7. Step 6: Changing Permissions
touch secure.txt
chmod 444 secure.txt
ls -l secure.txt   # verify

# 8. Step 7: Working with Environment Variables
export MY_VAR="Hello, Linux!"
echo $MY_VAR

# 9. Organize assignment in parent folder
cd ~
mv unzipped_dir devops-assignments/assignment-1

# 10. Optional: Save all commands in a text file
cd ~/devops-assignments/assignment-1
nano commands-used.txt
# (paste all commands, save with Ctrl+O → Enter → Ctrl+X)

# 11. Initialize Git repo and push to GitHub
cd ~/devops-assignments
git init
git add .
git commit -m "Add Assignment 1"
git branch -M main
git remote add origin <your_repo_url>
git push -u origin main
